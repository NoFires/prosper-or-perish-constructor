TRY_REPLACE:equality = {
	country_modifier = {
		monthly_towards_communalism = societal_value_monthly_move
	}

	priority = {
		value = 1 #all treated equally
		if = {
			limit = { has_tag = pp_food_security_priority }

			# Food storage needs highest priority to stop fluctuations in food for AI.
			if = {
				limit = { has_tag = pp_food_storage_priority }
				add = 120
			}
			# Direct food production needs second highest priority so we do not enter starvation loops.
			else_if = {
				limit = { has_tag = pp_direct_food_priority }
				add = 110
			}
			# Victuals markets need priority below cookeries so prepared food is distributed after direct food production.
			else_if = {
				limit = { has_tag = pp_food_distribution_priority }
				add = 100
			}
			# Irrigation and other water-control buildings need high priority so food production or capacity are not destroyed through underemployment.
			else_if = {
				limit = { has_tag = pp_water_control_priority }
				add = 95
			}
			# Staple-food producer buildings need to be manned first to avoid being outcompeted by non-food-related buildings.
			else_if = {
				limit = { has_tag = pp_staple_food_priority }
				add = 90
			}
		}
	}

	ai_will_do = {
		add = {
			desc = "inertia"
			value = -10
		}

		add = {
			desc = "societal_values"
			value = societal_value:individualism_vs_communalism
		}
	}
}
