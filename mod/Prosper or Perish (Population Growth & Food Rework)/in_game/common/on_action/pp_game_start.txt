# Zero main raw-material RGO capacity (max RGO workers) after history load; only where a raw good exists and cap > 0.
pp_reset_rgo_max_workers = {
	effect = {
		every_location_in_the_world = {
			limit = {
				exists = raw_material
				max_rgo_workers > 0
			}
			change_max_raw_material_workers = {
				value = max_rgo_workers
				multiply = -1
			}
		}
	}
}

# Game-start tweak: lower max RGO workers per raw good. Branch on `rgo_workers` (same as RGO level / raw-material panel; see pp_rgo_level_for_map); apply cuts with `change_max_raw_material_workers`.
pp_reduce_specific_rgo = {
	effect = {
		every_location_in_the_world = {
			limit = { exists = raw_material }
			# wheat — level >= 4: -2; level == 3: -1 (nested if/else_if so the chain cannot bind to another good's `if`)
			if = {
				limit = { raw_material = goods:wheat }
				if = {
					limit = { rgo_workers >= 4 }
					change_max_raw_material_workers = -2
				}
				else_if = {
					limit = { rgo_workers >= 3 }
					change_max_raw_material_workers = -1
				}
			}
			# legumes — apply when level >= 3; reduce max by 1
			if = {
				limit = {
					raw_material = goods:legumes
					rgo_workers >= 3
				}
				change_max_raw_material_workers = -1
			}
			# livestock — same nested pattern as wheat
			if = {
				limit = { raw_material = goods:livestock }
				if = {
					limit = { rgo_workers >= 4 }
					change_max_raw_material_workers = -2
				}
				else_if = {
					limit = { rgo_workers >= 3 }
					change_max_raw_material_workers = -1
				}
			}
			# rice — apply when level >= 2; reduce max by 1
			if = {
				limit = {
					raw_material = goods:rice
					rgo_workers >= 2
				}
				change_max_raw_material_workers = -1
			}
			# fish — apply when level >= 2; reduce max by 1
			if = {
				limit = {
					raw_material = goods:fish
					rgo_workers >= 2
				}
				change_max_raw_material_workers = -1
			}
			# wool — apply when level >= 2; reduce max by 1
			if = {
				limit = {
					raw_material = goods:wool
					rgo_workers >= 2
				}
				change_max_raw_material_workers = -1
			}
			# wild game — apply when level >= 2; reduce max by 1
			if = {
				limit = {
					raw_material = goods:wild_game
					rgo_workers >= 2
				}
				change_max_raw_material_workers = -1
			}
			if = {
				limit = {
					raw_material = goods:millet
					rgo_workers >= 3
				}
				change_max_raw_material_workers = -1
			}
			# cotton — apply when level >= 3; reduce max by 1
			if = {
				limit = {
					raw_material = goods:cotton
					rgo_workers >= 3
				}
				change_max_raw_material_workers = -1
			}
		}
	}
}

# Flags the welcome situation to spawn on the next monthly situation check (once per campaign).
# Do not re-set pending while the situation is already active (e.g. loading a save mid year-one).
pp_mod_welcome_situation_game_start = {
	effect = {
		if = {
			limit = {
				NOT = { has_global_variable = pp_mod_welcome_situation_completed }
				situation:pp_mod_welcome_situation = {
					situation_is_active = no
				}
			}
			set_global_variable = {
				name = pp_mod_welcome_situation_pending
				value = yes
			}
		}
	}
}

# Apply is_adjacent_to_lake modifier to locations that border lakes (engine does not auto-apply this like river_flowing_through/coastal)
pp_apply_adjacent_to_lake_modifier = {
	effect = {
		every_location_in_the_world = {
			limit = { is_adjacent_to_lake = yes }
			add_location_modifier = {
				modifier = is_adjacent_to_lake
				months = -1
				mode = replace
			}
		}
		every_location_in_the_world = {
			limit = {
				is_adjacent_to_lake = no
				has_location_modifier = is_adjacent_to_lake
			}
			remove_location_modifier = is_adjacent_to_lake
		}
	}
}

pp_precalculate_building_capacities = {
	effect = {
		every_location_in_the_world = {
			set_variable = { name = pp_farm_base_capacity value = 0 }
			set_variable = { name = pp_fish_base_capacity value = 0 }
			set_variable = { name = pp_forest_base_capacity value = 0 }
		}
		every_location_in_the_world = {
			limit = { is_ownable = yes }
			set_variable = {
				name = pp_farm_base_capacity
				value = pp_farm_base_capacity_value
			}
			set_variable = {
				name = pp_fish_base_capacity
				value = pp_fish_base_capacity_value
			}
			set_variable = {
				name = pp_forest_base_capacity
				value = pp_forest_base_capacity_value
			}
		}
	}
}

# Remove all buildings that cannot exist under mod rules (Option B).
# Runs after precalculate so static capacity baseline variables are available.
pp_remove_invalid_buildings = {
	effect = {
		every_location_in_the_world = {
			# Avoid changing buildings on unowned locations (engine can log null-owner Market Village).
			limit = { has_owner = yes }
			# Fruit orchard family: allowed in orchard-friendly farm locations.
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:fruit_orchard
						value > 0
					}
					NOT = {
						OR = {
							pp_orchard_friendly_location > 0
							pp_vanilla_start_fruit_orchard_location = yes
						}
					}
				}
				change_building_level_in_location = {
					building = building_type:fruit_orchard
					value = {
						value = "location_building_level(building_type:fruit_orchard)"
						multiply = -1
					}
				}
			}
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:pomological_orchard
						value > 0
					}
					NOT = { pp_orchard_friendly_location > 0 }
				}
				change_building_level_in_location = {
					building = building_type:pomological_orchard
					value = {
						value = "location_building_level(building_type:pomological_orchard)"
						multiply = -1
					}
				}
			}
			# Market village: reduce to rural_building_cap
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:market_village
						value > 0
					}
					location_building_level = {
						building_type = building_type:market_village
						value > rural_building_cap
					}
				}
				change_building_level_in_location = {
					building = building_type:market_village
					value = {
						value = rural_building_cap
						add = {
							value = "location_building_level(building_type:market_village)"
							multiply = -1
						}
					}
				}
			}
			# Fishing village: freshwater access or a fish RGO.
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:fishing_village
						value > 0
					}
					NOT = {
						OR = {
							has_river = yes
							is_adjacent_to_lake = yes
							topography = wetlands
							is_coastal = yes
							raw_material = goods:fish
						}
					}
				}
				change_building_level_in_location = {
					building = building_type:fishing_village
					value = {
						value = "location_building_level(building_type:fishing_village)"
						multiply = -1
					}
				}
			}
			# Ocean fishery: only allowed on coastal locations
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:ocean_fishery
						value > 0
					}
					NOT = { is_coastal = yes }
				}
				change_building_level_in_location = {
					building = building_type:ocean_fishery
					value = {
						value = "location_building_level(building_type:ocean_fishery)"
						multiply = -1
					}
				}
			}
			# Offshore fishery: only allowed on coastal locations
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:offshore_fishery
						value > 0
					}
					NOT = { is_coastal = yes }
				}
				change_building_level_in_location = {
					building = building_type:offshore_fishery
					value = {
						value = "location_building_level(building_type:offshore_fishery)"
						multiply = -1
					}
				}
			}
			# Farming village: only allowed in general farmable food locations.
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:farming_village
						value > 0
					}
					NOT = { pp_general_farmable_food_location > 0 }
				}
				change_building_level_in_location = {
					building = building_type:farming_village
					value = {
						value = "location_building_level(building_type:farming_village)"
						multiply = -1
					}
				}
			}
			# Sheep farms family: allowed in pasture-friendly locations.
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:sheep_farms
						value > 0
					}
					NOT = { pp_pasture_friendly_location > 0 }
				}
				change_building_level_in_location = {
					building = building_type:sheep_farms
					value = {
						value = "location_building_level(building_type:sheep_farms)"
						multiply = -1
					}
				}
			}
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:enclosed_sheep_walks
						value > 0
					}
					NOT = { pp_pasture_friendly_location > 0 }
				}
				change_building_level_in_location = {
					building = building_type:enclosed_sheep_walks
					value = {
						value = "location_building_level(building_type:enclosed_sheep_walks)"
						multiply = -1
					}
				}
			}
			# Forest buildings: forest/woods/jungle terrain or a forest RGO.
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:forest_village
						value > 0

						building_type = building_type:ocean_fishery
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}

				if = {
					limit = {
						population > 74
						raw_material = goods:fish
						NOT = { is_coastal = yes }
						fish_capacity_available > 0
						NOT = { has_building = building_type:fishing_village }
						NOT = { has_building = building_type:ocean_fishery }
						can_build_building = building_type:fishing_village
					}
					construct_building = {
						building_type = building_type:fishing_village
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}


				if = {
					limit = {
						population > 74
						forest_capacity_available > 0
						OR = {
							raw_material = goods:wild_game
							raw_material = goods:fur
							raw_material = goods:lumber
						}
						can_build_building = building_type:forest_village
						NOT = { has_building = building_type:forest_village }
					}
					construct_building = {
						building_type = building_type:forest_village
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}

			}
		}
	}
