pp_fish_base_capacity_value = {
	value = 0
	if = {
		limit = {
			is_ownable = yes
			raw_material = goods:fish
		}
		add = 3.00
	}
	if = {
		limit = { is_coastal = yes }
		add = 4.50
	}
	if = {
		limit = { is_adjacent_to_lake = yes }
		add = 3.00
	}
	if = {
		limit = { has_river = yes }
		add = 2.25
	}
	if = {
		limit = { topography = wetlands }
		add = 1.50
	}
}

pp_forest_base_capacity_value = {
	value = 0
	if = {
		limit = {
			is_ownable = yes
			OR = {
				raw_material = goods:lumber
				raw_material = goods:fur
				raw_material = goods:wild_game
			}
		}
		add = 4
	}
	if = {
		limit = { vegetation = forest }
		add = 6
	}
	else_if = {
		limit = { vegetation = woods }
		add = 4
	}
	else_if = {
		limit = { vegetation = jungle }
		add = 3
	}
}
