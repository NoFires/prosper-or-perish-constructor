			desc = "BUILDING_LEVEL_RGO_SIZE_FARMING"
			value = var:pp_farm_base_capacity
			multiply = max_rgo_workers
			multiply = 0.125
		}
	}
}

	add = {
		desc = "BUILDING_LEVEL_BASE_FARM_RGO"

			limit = { has_variable = pp_farm_base_capacity }

		else = { value = 0 }

	}
	add = {
		desc = "BUILDING_LEVEL_FARM_CAPACITY_IMPROVEMENTS"

	}
}
