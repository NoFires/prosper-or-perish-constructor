cookery_max_level = {
	add = {
		desc = "BUILDING_LEVEL_BASE"
		value = 1
	}
	add = {
		desc = "BUILDING_LEVEL_DEVELOPMENT"
		value = development
		multiply = 0.20
	}
	add = {
		desc = "BUILDING_LEVEL_POPULATION"
		value = population
		multiply = 0.025
	}
	if = {
		limit = {
			location_rank = location_rank:city
		}
		add = {
			desc = "BUILDING_LEVEL_IS_CITY"
			value = 3
		}
	}
}
farm_max_level = {
	add = {
		desc = "BUILDING_LEVEL_RGO_SIZE_FARMING"
		value = max_rgo_workers
		multiply = 0.75
	}
	add = {
		desc = "BUILDING_LEVEL_POPULATION_CAPACITY_FARMING"
		value = modifier:local_population_capacity
		multiply = 0.08
	}
	add = {
		desc = "BUILDING_LEVEL_FROM_LOCATION_RANK_FARMING"
		value = modifier:farm_max_level_modifier
	}
	min = 0
}

farm_capacity_remaining = {
	add = farm_max_level
	add = {
		desc = "BUILDING_LEVEL_AVAILABLE_SPACE"
		value = total_building_levels
		multiply = -0.1
	}
	add = {
		desc = "BUILDING_LEVEL_FARM_SPACE_USED"
		value = modifier:farm_space_used
		multiply = -1
	}
}

farm_capacity_available = {
	add = farm_capacity_remaining
	min = 0
}
