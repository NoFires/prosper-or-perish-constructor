pp_food_security_building_priority = {
	value = 0

	# Food storage needs highest priority to stop fluctuations in food for AI.
	if = {
		limit = { building_type = building_type:granary }
		add = 120
	}

	# Direct food production needs second highest priority so we do not enter starvation loops.
	if = {
		limit = {
			OR = {
				building_type = building_type:cookery
				building_type = building_type:victualling_yard
			}
		}
		add = 110
	}

	# Irrigation and other water-control buildings need high priority so food production or capacity are not destroyed through underemployment.
	if = {
		limit = {
			OR = {
				building_type = building_type:irrigation_systems
				building_type = building_type:bund
				building_type = building_type:terraces
				building_type = building_type:polders
				building_type = building_type:khmer_baray
			}
		}
		add = 95
	}

	# Staple-food producer buildings need to be manned first to avoid being outcompeted by non-food-related buildings.
	if = {
		limit = {
			OR = {
				building_type = building_type:farming_village
				building_type = building_type:farming_village_rotations
				building_type = building_type:model_farm
				building_type = building_type:fishing_village
				building_type = building_type:ocean_fishery
				building_type = building_type:offshore_fishery
				building_type = building_type:fruit_orchard
				building_type = building_type:pomological_orchard
				building_type = building_type:forest_village
				building_type = building_type:managed_forest_village
				building_type = building_type:sheep_farms
				building_type = building_type:enclosed_sheep_walks
			}
		}
		add = 90
	}
}
