	add = {
		desc = "BUILDING_LEVEL_BASE"
		value = 1
	}
	add = {
		desc = "BUILDING_LEVEL_DEVELOPMENT"
		value = development
		multiply = 0.15
	}
	add = {
		desc = "BUILDING_LEVEL_RGO_MAX"
		value = max_rgo_workers
		multiply = 0.65
	}
	if = {
		limit = {
			has_river = yes
		}

			desc = "BUILDING_LEVEL_HAS_RIVER"
			value = 1
		}
	}
}
