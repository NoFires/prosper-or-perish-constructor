# Customizable Localization for Building Capacity Breakdowns
# Used to display individual bonuses in map mode tooltips since direct data calls like GetClimate.GetKey are broken in the data system.


# --- FISHING VILLAGE ---

pp_fishing_water_breakdown = {
	type = location
	text = {
		localization_key = PP_FISHING_COASTAL_LAKE_RIVER
		trigger = {
			is_coastal = yes
			is_adjacent_to_lake = yes
			has_river = yes
		}
	}
	text = {
		localization_key = PP_FISHING_COASTAL_LAKE
		trigger = {
			is_coastal = yes
			is_adjacent_to_lake = yes
		}
	}
	text = {
		localization_key = PP_FISHING_COASTAL_RIVER
		trigger = {
			is_coastal = yes
			has_river = yes
		}
	}
	text = {
		localization_key = PP_FISHING_LAKE_RIVER
		trigger = {
			is_adjacent_to_lake = yes
			has_river = yes
		}
	}
	text = {
		localization_key = PP_FISHING_COASTAL
		trigger = { is_coastal = yes }
	}
	text = {
		localization_key = PP_FISHING_LAKE
		trigger = { is_adjacent_to_lake = yes }
	}
	text = {
		localization_key = PP_FISHING_RIVER
		trigger = { has_river = yes }
	}
	text = {
		localization_key = PP_EMPTY_LOC
	}
}
