pp_market_food_price = {
	map_color = {
		if = {
			limit = {
				has_owner = yes
				market.food_price < @pp_market_food_price_very_cheap
			}
			value = define:NMapColors|MAP_COLOR_MAX
		}
		else_if = {
			limit = {
				has_owner = yes
				market.food_price < @pp_market_food_price_cheap
			}
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_MAX
				max_color = define:NMapColors|MAP_COLOR_HIGH
				factor = {
					value = market.food_price
					subtract = @pp_market_food_price_very_cheap
					divide = 0.04
					max = 1
					min = 0
				}
			}
		}
		else_if = {
			limit = {
				has_owner = yes
				market.food_price < @pp_market_food_price_neutral
			}
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_HIGH
				max_color = define:NMapColors|MAP_COLOR_MID
				factor = {
					value = market.food_price
					subtract = @pp_market_food_price_cheap
					divide = 0.066
					max = 1
					min = 0
				}
			}
		}
		else_if = {
			limit = {
				has_owner = yes
				market.food_price < @pp_market_food_price_expensive
			}
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_MID
				max_color = define:NMapColors|MAP_COLOR_LOW
				factor = {
					value = market.food_price
					subtract = @pp_market_food_price_neutral
					divide = 0.051
					max = 1
					min = 0
				}
			}
		}
		else_if = {
			limit = {
				has_owner = yes
				market.food_price < @pp_market_food_price_severe
			}
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_LOW
				max_color = define:NMapColors|MAP_COLOR_MIN
				factor = {
					value = market.food_price
					subtract = @pp_market_food_price_expensive
					divide = 0.291
					max = 1
					min = 0
				}
			}
		}
		else_if = {
			limit = { has_owner = yes }
			value = define:NMapColors|MAP_COLOR_MIN
		}
		else = {
			value = rgb { 128 128 128 }
		}
	}

	legend_key = {
		desc = "MAPMODE_PP_MARKET_FOOD_PRICE_VERY_CHEAP"
		color = define:NMapColors|MAP_COLOR_MAX
	}
	legend_key = {
		desc = "MAPMODE_PP_MARKET_FOOD_PRICE_CHEAP"
		color = define:NMapColors|MAP_COLOR_HIGH
	}
	legend_key = {
		desc = "MAPMODE_PP_MARKET_FOOD_PRICE_NEUTRAL"
		color = define:NMapColors|MAP_COLOR_MID
	}
	legend_key = {
		desc = "MAPMODE_PP_MARKET_FOOD_PRICE_EXPENSIVE"
		color = define:NMapColors|MAP_COLOR_LOW
	}
	legend_key = {
		desc = "MAPMODE_PP_MARKET_FOOD_PRICE_SEVERE"
		color = define:NMapColors|MAP_COLOR_MIN
	}

	tooltip_key = {
		if = { limit = { has_owner = yes } value = MAPMODE_PP_MARKET_FOOD_PRICE_TT_LAND }
		else = { value = MAPMODE_PP_MARKET_FOOD_PRICE_TT_WATER }
	}

	small_map_names = market
	medium_map_names = market
	large_map_names = market

	small_tooltip_context = market
	medium_tooltip_context = market
	large_tooltip_context = market

	fill_in_impassable = no
	enable_snow = no
	enable_water_names = yes

	flatmap_behaviour = Always
	use_fow = no

	category = economy
	index = 3

	map_markers = {
		all = no
		market_marker = yes
		toll_marker = yes
	}

	gradient_parameters = {
		zoom_step = @zoom_step_near
		gradient_alpha_inside = 	@data_gradient_alpha_inside
		gradient_alpha_outside = 	@data_gradient_alpha_outside
		gradient_width = 			@data_gradient_width
		gradient_color_mult =		@data_gradient_color_mult
		edge_width = 				@data_edge_width
		edge_sharpness = 			@data_edge_sharpness
		edge_alpha = 				@data_edge_alpha
		edge_color_mult = 			@data_edge_color_mult
		before_lighting_blend =		@data_before_lighting_blend
		after_lighting_blend =		@data_after_lighting_blend
	}

	refresh_colors_on_selection_change = yes
	color_refresh_counters = { Month }
	color_and_names_refresh_counters = { MarketReach LocationOwnerChanged }
	map_lines_mode = ToMarketCenter
}
