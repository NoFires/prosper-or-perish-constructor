# Map mode: total building levels per location (traffic-light bucketed heatmap).
# Buckets are tuned from current savegame building-level distributions.

@pp_building_levels_cap = 93

@zoom_step_near = 2
@data_gradient_alpha_inside = 1
@data_gradient_alpha_outside = 1
@data_gradient_width = 0.25
@data_gradient_color_mult = 0.9
@data_edge_width = 0
@data_edge_sharpness = 0.01
@data_edge_alpha = 0
@data_edge_color_mult = 0
@data_before_lighting_blend = 0.5
@data_after_lighting_blend = 0.5

pp_building_levels = {
	map_color = {
		if = {
			limit = { is_ownable = no }
			value = rgb { 128 128 128 }
		}
		else_if = {
			limit = { has_owner = yes }
			if = {
				limit = { total_building_levels < 3 }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_MIN
					max_color = define:NMapColors|MAP_COLOR_LOW
					factor = { value = total_building_levels divide = 3 max = 1 min = 0 }
				}
			}
			else_if = {
				limit = { total_building_levels < 8 }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_LOW
					max_color = define:NMapColors|MAP_COLOR_MID
					factor = { value = total_building_levels subtract = 3 divide = 5 max = 1 min = 0 }
				}
			}
			else_if = {
				limit = { total_building_levels < 22 }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_MID
					max_color = define:NMapColors|MAP_COLOR_HIGH
					factor = { value = total_building_levels subtract = 8 divide = 14 max = 1 min = 0 }
				}
			}
			else_if = {
				limit = { total_building_levels < @pp_building_levels_cap }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_HIGH
					max_color = define:NMapColors|MAP_COLOR_MAX
					factor = { value = total_building_levels subtract = 22 divide = 71 max = 1 min = 0 }
				}
			}
			else = { value = define:NMapColors|MAP_COLOR_MAX }
		}
		else = {
			value = rgb { 128 128 128 }
		}
	}
	legend_key = { desc = "MAPMODE_PP_BUILDING_LEVELS_VERY_LOW" color = define:NMapColors|MAP_COLOR_MIN }
	legend_key = { desc = "MAPMODE_PP_BUILDING_LEVELS_LOW" color = define:NMapColors|MAP_COLOR_LOW }
	legend_key = { desc = "MAPMODE_PP_BUILDING_LEVELS_MEDIUM" color = define:NMapColors|MAP_COLOR_MID }
	legend_key = { desc = "MAPMODE_PP_BUILDING_LEVELS_HIGH" color = define:NMapColors|MAP_COLOR_HIGH }
	legend_key = { desc = "MAPMODE_PP_BUILDING_LEVELS_CAPPED" color = define:NMapColors|MAP_COLOR_MAX }
	tooltip_key = {
		if = { limit = { is_land = yes } value = MAPMODE_PP_BUILDING_LEVELS_TT_LAND }
		else = { value = MAPMODE_PP_BUILDING_LEVELS_TT_WATER }
	}
	small_map_names = location
	medium_map_names = location
	large_map_names = location
	small_tooltip_context = location
	medium_tooltip_context = location
	large_tooltip_context = location
	fill_in_impassable = yes
	enable_snow = no
	flatmap_behaviour = always
	use_fow = no
	category = economy
	index = 0
	map_markers = { all = no }
	gradient_parameters = {
		zoom_step = @zoom_step_near
		gradient_alpha_inside = @data_gradient_alpha_inside
		gradient_alpha_outside = @data_gradient_alpha_outside
		gradient_width = @data_gradient_width
		gradient_color_mult = @data_gradient_color_mult
		edge_width = @data_edge_width
		edge_sharpness = @data_edge_sharpness
		edge_alpha = @data_edge_alpha
		edge_color_mult = @data_edge_color_mult
		before_lighting_blend = @data_before_lighting_blend
		after_lighting_blend = @data_after_lighting_blend
	}
	color_refresh_counters = { ProductionList LocationDevelopmentChanged }
	color_and_names_refresh_counters = { LocationOwnerChanged CountryStatus }
}
