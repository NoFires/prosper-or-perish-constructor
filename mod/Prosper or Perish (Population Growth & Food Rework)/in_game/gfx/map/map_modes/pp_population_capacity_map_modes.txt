# Map mode for population capacity per location.
# Traffic-light sequential scale: red = low, yellow = middle, green = high. Uncolonized ownable land is included.
# Uses modifier:local_population_capacity in the generated in-game unit.

@zoom_step_near = 2
@data_gradient_alpha_inside = 1
@data_gradient_alpha_outside = 1
@data_gradient_width = 0.25
@data_gradient_color_mult = 0.9
@data_edge_width = 0
@data_edge_sharpness = 0.01
@data_edge_alpha = 0
@data_edge_color_mult = 0
@data_before_lighting_blend = 0.5
@data_after_lighting_blend = 0.5
@pp_population_capacity_map_cap = 66
@pp_population_growth_negative_cap = -0.007
@pp_population_growth_negative_mid = -0.002
@pp_population_growth_neutral_low = -0.0005
@pp_population_growth_neutral_high = 0.0005
@pp_population_growth_positive_mid = 0.00375

# ~0.9% monthly local growth modifier (script fraction); stripe marks values at or above the display cap.
@pp_population_growth_cap_stripe = 0.009

pp_population_capacity = {
	map_color = {
		if = {
			limit = { is_ownable = no }
			value = rgb { 128 128 128 }
		}
		else_if = {
			limit = {
				modifier:local_population_capacity < 13
			}
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_MIN
				max_color = define:NMapColors|MAP_COLOR_LOW
				factor = {
					value = modifier:local_population_capacity
					divide = 13
					max = 1
					min = 0
				}
			}
		}
		else_if = {
			limit = {
				modifier:local_population_capacity < 36
			}
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_LOW
				max_color = define:NMapColors|MAP_COLOR_MID
				factor = {
					value = modifier:local_population_capacity
					subtract = 13
					divide = 23
					max = 1
					min = 0
				}
			}
		}
		else_if = {
			limit = {
				modifier:local_population_capacity < 55
			}
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_MID
				max_color = define:NMapColors|MAP_COLOR_HIGH
				factor = {
					value = modifier:local_population_capacity
					subtract = 36
					divide = 19
					max = 1
					min = 0
				}
			}
		}
		else_if = {
			limit = {
				modifier:local_population_capacity < @pp_population_capacity_map_cap
			}
			lerp = {
				min_color = define:NMapColors|MAP_COLOR_HIGH
				max_color = define:NMapColors|MAP_COLOR_MAX
				factor = {
					value = modifier:local_population_capacity
					subtract = 55
					divide = 11
					max = 1
					min = 0
				}
			}
		}
		else = {
			value = define:NMapColors|MAP_COLOR_MAX
		}
	}
	legend_key = { desc = "MAPMODE_PP_POPULATION_CAPACITY_VERY_LOW" color = define:NMapColors|MAP_COLOR_MIN }
	legend_key = { desc = "MAPMODE_PP_POPULATION_CAPACITY_LOW" color = define:NMapColors|MAP_COLOR_LOW }
	legend_key = { desc = "MAPMODE_PP_POPULATION_CAPACITY_MEDIUM" color = define:NMapColors|MAP_COLOR_MID }
	legend_key = { desc = "MAPMODE_PP_POPULATION_CAPACITY_HIGH" color = define:NMapColors|MAP_COLOR_HIGH }
	legend_key = { desc = "MAPMODE_PP_POPULATION_CAPACITY_CAPPED" color = define:NMapColors|MAP_COLOR_MAX }
	tooltip_key = {
		if = { limit = { is_land = yes } value = MAPMODE_PP_POPULATION_CAPACITY_TT_LAND }
		else = { value = MAPMODE_PP_POPULATION_CAPACITY_TT_WATER }
	}
	small_map_names = location
	medium_map_names = location
	large_map_names = location
	small_tooltip_context = location
	medium_tooltip_context = location
	large_tooltip_context = location
	fill_in_impassable = yes
	enable_snow = no
	flatmap_behaviour = always
	use_fow = no
	category = population
	index = 1
	map_markers = { all = no }
	gradient_parameters = {
		zoom_step = @zoom_step_near
		gradient_alpha_inside = @data_gradient_alpha_inside
		gradient_alpha_outside = @data_gradient_alpha_outside
		gradient_width = @data_gradient_width
		gradient_color_mult = @data_gradient_color_mult
		edge_width = @data_edge_width
		edge_sharpness = @data_edge_sharpness
		edge_alpha = @data_edge_alpha
		edge_color_mult = @data_edge_color_mult
		before_lighting_blend = @data_before_lighting_blend
		after_lighting_blend = @data_after_lighting_blend
	}
	color_refresh_counters = { LocationDevelopmentChanged LocationPopulationChanged }
	color_and_names_refresh_counters = { LocationOwnerChanged CountryStatus }
}

# Map mode for local population growth modifier per location.
# Diverging traffic-light heatmap for local population growth: red = negative, yellow = near zero, green = positive.
# Stripes (secondary): local modifier at or above the display cap first, then starving locations.

pp_population_growth = {
	map_color = {
		if = {
			limit = { is_ownable = no }
			value = rgb { 128 128 128 }
		}
		else_if = {
			limit = { has_owner = yes }
			if = {
				limit = { modifier:local_population_growth < @pp_population_growth_negative_cap }
				value = define:NMapColors|MAP_COLOR_MIN
			}
			else_if = {
				limit = { modifier:local_population_growth < @pp_population_growth_negative_mid }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_MIN
					max_color = define:NMapColors|MAP_COLOR_LOW
					factor = {
						value = modifier:local_population_growth
						add = 0.007
						divide = 0.005
						max = 1
						min = 0
					}
				}
			}
			else_if = {
				limit = { modifier:local_population_growth < @pp_population_growth_neutral_low }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_LOW
					max_color = define:NMapColors|MAP_COLOR_MID
					factor = {
						value = modifier:local_population_growth
						add = 0.002
						divide = 0.0015
						max = 1
						min = 0
					}
				}
			}
			else_if = {
				limit = { modifier:local_population_growth < @pp_population_growth_neutral_high }
				value = define:NMapColors|MAP_COLOR_MID
			}
			else_if = {
				limit = { modifier:local_population_growth < @pp_population_growth_positive_mid }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_MID
					max_color = define:NMapColors|MAP_COLOR_HIGH
					factor = {
						value = modifier:local_population_growth
						subtract = 0.0005
						divide = 0.00325
						max = 1
						min = 0
					}
				}
			}
			else_if = {
				limit = { modifier:local_population_growth < @pp_population_growth_cap_stripe }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_HIGH
					max_color = define:NMapColors|MAP_COLOR_MAX
					factor = {
						value = modifier:local_population_growth
						subtract = @pp_population_growth_positive_mid
						divide = 0.00525
						max = 1
						min = 0
					}
				}
			}
			else = {
				value = define:NMapColors|MAP_COLOR_MAX
			}
		}
		else = {
			value = rgb { 128 128 128 }
		}
	}
	secondary_map_color = {
		if = {
			limit = {
				has_owner = yes
				modifier:local_population_growth >= @pp_population_growth_cap_stripe
			}
			value = rgb { 35 35 42 }
		}
		else_if = {
			limit = {
				has_owner = yes
				province = { is_starving = yes }
			}
			value = define:NMapColors|POPULATION_STARVING_COLOR_STRIPE
		}
	}
	legend_key = { desc = "MAPMODE_PP_POPULATION_GROWTH_EXTREME_DEFICIT" color = define:NMapColors|MAP_COLOR_MIN }
	legend_key = { desc = "MAPMODE_PP_POPULATION_GROWTH_DEFICIT" color = define:NMapColors|MAP_COLOR_LOW }
	legend_key = { desc = "MAPMODE_PP_POPULATION_GROWTH_NEUTRAL" color = define:NMapColors|MAP_COLOR_MID }
	legend_key = { desc = "MAPMODE_PP_POPULATION_GROWTH_GOOD" color = define:NMapColors|MAP_COLOR_HIGH }
	legend_key = { desc = "MAPMODE_PP_POPULATION_GROWTH_EXCEPTIONAL" color = define:NMapColors|MAP_COLOR_MAX }
	legend_key = { desc = "MAPMODE_PP_POPULATION_GROWTH_STARVING" color = define:NMapColors|POPULATION_STARVING_COLOR_STRIPE }
	legend_key = { desc = "MAPMODE_PP_POPULATION_GROWTH_STRIPE" color = rgb { 35 35 42 } }
	tooltip_key = {
		if = { limit = { is_land = yes } value = MAPMODE_PP_POPULATION_GROWTH_TT_LAND }
		else = { value = MAPMODE_PP_POPULATION_GROWTH_TT_WATER }
	}
	small_map_names = location
	medium_map_names = location
	large_map_names = location
	small_tooltip_context = location
	medium_tooltip_context = location
	large_tooltip_context = location
	fill_in_impassable = yes
	enable_snow = no
	flatmap_behaviour = always
	use_fow = no
	category = population
	index = 1
	map_markers = { all = no }
	gradient_parameters = {
		zoom_step = @zoom_step_near
		gradient_alpha_inside = @data_gradient_alpha_inside
		gradient_alpha_outside = @data_gradient_alpha_outside
		gradient_width = @data_gradient_width
		gradient_color_mult = @data_gradient_color_mult
		edge_width = @data_edge_width
		edge_sharpness = @data_edge_sharpness
		edge_alpha = @data_edge_alpha
		edge_color_mult = @data_edge_color_mult
		before_lighting_blend = @data_before_lighting_blend
		after_lighting_blend = @data_after_lighting_blend
	}
	color_refresh_counters = { LocationDevelopmentChanged LocationPopulationChanged }
	color_and_names_refresh_counters = { LocationOwnerChanged CountryStatus }
}
